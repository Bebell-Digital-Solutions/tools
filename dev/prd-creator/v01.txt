    <script>
        // Initialize Lucide Icons
        lucide.createIcons();

        // State Management
        let features = ["User authentication and profile management", "Dashboard analytics"];
        
        // DOM Elements
        const inputs = {
            title: document.getElementById('input-title'),
            description: document.getElementById('input-description'),
            objectives: document.getElementById('input-objectives'),
            audience: document.getElementById('input-audience'),
            color1: document.getElementById('input-color-1'),
            color2: document.getElementById('input-color-2'),
            color3: document.getElementById('input-color-3'),
            icons: document.getElementById('input-icons'),
            language: document.getElementById('input-language'),
            theme: document.getElementById('input-theme'),
            platform: document.getElementById('input-platform'),
            frontend: document.getElementById('input-frontend'),
            backend: document.getElementById('input-backend'),
            hosting: document.getElementById('input-hosting'),
            date: document.getElementById('input-date'),
            stakeholders: document.getElementById('input-stakeholders'),
        };

        const previews = {
            title: document.getElementById('prev-title'),
            description: document.getElementById('prev-description'),
            objectives: document.getElementById('prev-objectives'),
            audience: document.getElementById('prev-audience'),
            features: document.getElementById('prev-features'),
            palette: document.getElementById('prev-palette'),
            icons: document.getElementById('prev-icons'),
            language: document.getElementById('prev-language'),
            theme: document.getElementById('prev-theme'),
            platform: document.getElementById('prev-platform'),
            frontend: document.getElementById('prev-frontend'),
            backend: document.getElementById('prev-backend'),
            hosting: document.getElementById('prev-hosting'),
            date: document.getElementById('prev-date'),
            stakeholders: document.getElementById('prev-stakeholders'),
        };

        // Event Listeners for Live Preview
        Object.values(inputs).forEach(input => {
            input.addEventListener('input', updatePreview);
            input.addEventListener('change', updatePreview);
        });

        // Initialize features
        renderFeaturesList();
        updatePreview();

        // Feature Management Functions
        function renderFeaturesList() {
            const container = document.getElementById('features-container');
            container.innerHTML = '';
            
            features.forEach((feat, index) => {
                const div = document.createElement('div');
                div.className = 'flex gap-2 group';
                div.innerHTML = `
                    <input type="text" value="${escapeHtml(feat)}" oninput="updateFeature(${index}, this.value)" placeholder="Describe feature..." class="input-transition flex-1 min-w-0 px-4 py-2 bg-gray-50 rounded-xl border border-gray-200 focus:outline-none focus:bg-white focus:ring-2 focus:ring-brandPrimary focus:border-transparent text-sm">
                    <button onclick="removeFeature(${index})" class="p-2 flex-shrink-0 text-gray-400 hover:text-red-500 hover:bg-red-50 rounded-lg transition-colors" title="Remove feature">
                        <i data-lucide="trash-2" class="w-5 h-5"></i>
                    </button>
                `;
                container.appendChild(div);
            });
            lucide.createIcons();
            updatePreview();
        }

        function addFeature() {
            features.push("");
            renderFeaturesList();
            
            // Focus new input
            const container = document.getElementById('features-container');
            const inputs = container.querySelectorAll('input');
            if(inputs.length > 0) {
                inputs[inputs.length - 1].focus();
            }
        }

        function addPresetFeature() {
            const select = document.getElementById('preset-features');
            if(select.value) {
                features.push(select.value);
                select.value = ""; // reset dropdown
                renderFeaturesList();
            }
        }

        function updateFeature(index, value) {
            features[index] = value;
            updatePreview();
        }

        function removeFeature(index) {
            features.splice(index, 1);
            renderFeaturesList();
        }

        // Main Update Preview Function
        function updatePreview() {
            // Text fields
            updateTextField(previews.title, inputs.title.value, 'Untitled Project');
            updateTextField(previews.description, inputs.description.value, 'No description provided.');
            updateTextField(previews.objectives, inputs.objectives.value, 'No objectives provided.');
            updateTextField(previews.audience, inputs.audience.value, 'Not specified.');
            
            // Features
            if (features.filter(f => f.trim() !== "").length === 0) {
                previews.features.innerHTML = '<li class="italic text-gray-400">No features added.</li>';
            } else {
                previews.features.innerHTML = features
                    .filter(f => f.trim() !== "")
                    .map(f => `<li>${escapeHtml(f)}</li>`)
                    .join('');
            }

            // Design & Tech
            previews.palette.innerHTML = `
                <div class="w-6 h-6 rounded-md shadow-sm border border-gray-200" style="background-color: ${inputs.color1.value}" title="${inputs.color1.value}"></div>
                <div class="w-6 h-6 rounded-md shadow-sm border border-gray-200" style="background-color: ${inputs.color2.value}" title="${inputs.color2.value}"></div>
                <div class="w-6 h-6 rounded-md shadow-sm border border-gray-200" style="background-color: ${inputs.color3.value}" title="${inputs.color3.value}"></div>
            `;
            previews.icons.innerText = inputs.icons.value;
            previews.theme.innerText = inputs.theme.value;

            previews.language.innerText = inputs.language.value;
            previews.platform.innerText = inputs.platform.value;

            updateTechField(previews.frontend, inputs.frontend.value);
            updateTechField(previews.backend, inputs.backend.value);
            updateTechField(previews.hosting, inputs.hosting.value);

            // Date formatting
            const dateVal = inputs.date.value;
            if (dateVal) {
                const dateObj = new Date(dateVal);
                previews.date.innerText = dateObj.toLocaleDateString(undefined, { year: 'numeric', month: 'long', day: 'numeric' });
                previews.date.className = 'col-span-2 font-medium text-gray-800';
            } else {
                previews.date.innerText = 'Not set';
                previews.date.className = 'col-span-2 italic text-gray-400';
            }

            // Stakeholders
            const stakeholdersVal = inputs.stakeholders.value;
            if (stakeholdersVal.trim() === '') {
                previews.stakeholders.innerHTML = '<li class="italic text-gray-400">None defined.</li>';
            } else {
                previews.stakeholders.innerHTML = stakeholdersVal
                    .split(',')
                    .map(s => s.trim())
                    .filter(s => s.length > 0)
                    .map(s => `<li>• ${escapeHtml(s)}</li>`)
                    .join('');
            }
        }

        // Helper for formatting preview text
        function updateTextField(element, value, placeholder) {
            if (value.trim() === '') {
                element.innerText = placeholder;
                element.classList.add('italic', 'text-gray-400');
                element.classList.remove('text-gray-900', 'text-gray-600');
            } else {
                element.innerText = value;
                element.classList.remove('italic', 'text-gray-400');
                if(element.tagName === 'H1') element.classList.add('text-gray-900');
                else element.classList.add('text-gray-600');
            }
        }

        function updateTechField(element, value) {
            if (value.trim() === '') {
                element.innerText = 'Not set';
                element.classList.add('text-gray-400', 'italic');
                element.classList.remove('text-gray-800');
            } else {
                element.innerText = value;
                element.classList.remove('text-gray-400', 'italic');
                element.classList.add('text-gray-800');
            }
        }

        // Utility to prevent XSS
        function escapeHtml(unsafe) {
            return unsafe
                 .replace(/&/g, "&amp;")
                 .replace(/</g, "&lt;")
                 .replace(/>/g, "&gt;")
                 .replace(/"/g, "&quot;")
                 .replace(/'/g, "&#039;");
        }

        // Actions
        function copyToClipboard() {
            const text = generatePlainText();
            
            // Method 1: Clipboard API (Requires HTTPS)
            if (navigator.clipboard && window.isSecureContext) {
                navigator.clipboard.writeText(text).then(() => {
                    showMessage("PRD Text Copied to Clipboard!");
                }).catch(err => {
                    // Fallback to execCommand
                    fallbackCopy(text);
                });
            } else {
                // Fallback for older browsers / iframe restrictions
                fallbackCopy(text);
            }
        }

        function fallbackCopy(text) {
            const textArea = document.createElement("textarea");
            textArea.value = text;
            textArea.style.position = "fixed"; 
            document.body.appendChild(textArea);
            textArea.focus();
            textArea.select();
            try {
                document.execCommand('copy');
                showMessage("PRD Text Copied to Clipboard!");
            } catch (err) {
                console.error('Fallback copy failed', err);
                alert("Failed to copy text. Please try again.");
            }
            document.body.removeChild(textArea);
        }

        function downloadPDF() {
            const element = document.getElementById('prd-document');
            const title = inputs.title.value ? inputs.title.value.replace(/[^a-z0-9]/gi, '_').toLowerCase() : 'untitled_project';
            const filename = `PRD_${title}.pdf`;
            
            const btn = document.getElementById('btn-pdf');
            const originalContent = btn.innerHTML;
            btn.innerHTML = `<i data-lucide="loader-2" class="w-4 h-4 animate-spin"></i> Generating...`;
            lucide.createIcons();
            
            const opt = {
                margin:       0.5,
                filename:     filename,
                image:        { type: 'jpeg', quality: 0.98 },
                html2canvas:  { scale: 2, useCORS: true, logging: false },
                jsPDF:        { unit: 'in', format: 'letter', orientation: 'portrait' }
            };

            html2pdf().set(opt).from(element).save().then(() => {
                btn.innerHTML = originalContent;
                lucide.createIcons();
                showMessage("PDF Downloaded Successfully!");
            }).catch(err => {
                console.error('PDF Generation Error:', err);
                btn.innerHTML = originalContent;
                lucide.createIcons();
                alert('Error generating PDF.');
            });
        }

        // Custom Message UI (Replacement for native alert)
        function showMessage(msg) {
            const modal = document.getElementById('message-modal');
            const content = document.getElementById('modal-content');
            document.getElementById('message-text').innerText = msg;
            
            modal.classList.remove('hidden');
            // Trigger reflow
            void modal.offsetWidth;
            
            modal.classList.remove('opacity-0');
            content.classList.remove('scale-95');
            content.classList.add('scale-100');

            setTimeout(() => {
                modal.classList.add('opacity-0');
                content.classList.remove('scale-100');
                content.classList.add('scale-95');
                setTimeout(() => {
                    modal.classList.add('hidden');
                }, 200); // match transition duration
            }, 2500);
        }
    </script>
